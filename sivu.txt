
!macro basic_start(addr) {
* = $801
    !byte $0b, $08, $0a, $00, $9e
    !for d in [10000, 1000, 100, 10, 1] {
        !if (addr >= d) {
            !byte $30 + (addr/d)%10
        }
    }
    !byte 0, 0, 0
}

+basic_start(entry)

entry:
    ldx $#0
loop:
    txa
    sta $d020
    sta $d021

    jmp loop
